中文转Unicode

String chinese = "我爱祖国";
StringBuffer unicode = new StringBuffer();
for (int i = 0; i < chinese.length(); i++) {
    Character character = chinese.charAt(i);
    unicode.append("\\u" + Integer.toHexString(character));
}

System.out.println(unicode.toString());
